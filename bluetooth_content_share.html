<?php
// Database configuration
$host = 'localhost';
$db_user = 'your_database_user';
$db_password = 'your_database_password';
$db_name = 'your_database_name';

// Connect to the database
$conn = new mysqli($host, $db_user, $db_password, $db_name);

// Check connection
if ($conn->connect_error) {
    die("Connection failed: " . $conn->connect_error);
}

// Initialize variables
$selectedForm = '';
$names = [];

// Handle form submission
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    // Get the selected option
    $selectedForm = $_POST['form_option'];

    // Fetch names from the database based on the selected form
    $sql = "SELECT name FROM your_table_name WHERE form_type = '$selectedForm'";
    $result = $conn->query($sql);

    if ($result && $result->num_rows > 0) {
        while ($row = $result->fetch_assoc()) {
            $names[] = $row['name'];
        }
    } else {
        $error = "No names found for the selected form.";
    }
}

$conn->close();
?>

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Dynamic Form Filter</title>
</head>
<body>
    <h1>Filter Names by Form</h1>

    <!-- Form to select the option -->
    <form method="POST">
        <label for="form_option">Select a Form:</label>
        <select name="form_option" id="form_option" required>
            <option value="">-- Select a Form --</option>
            <option value="form1" <?= $selectedForm === 'form1' ? 'selected' : '' ?>>Form 1</option>
            <option value="form2" <?= $selectedForm === 'form2' ? 'selected' : '' ?>>Form 2</option>
            <option value="form3" <?= $selectedForm === 'form3' ? 'selected' : '' ?>>Form 3</option>
            <option value="form4" <?= $selectedForm === 'form4' ? 'selected' : '' ?>>Form 4</option>
        </select>
        <button type="submit">Filter</button>
    </form>

    <!-- Display results -->
    <h2>Results:</h2>
    <?php if (!empty($names)): ?>
        <ul>
            <?php foreach ($names as $name): ?>
                <li><?= htmlspecialchars($name) ?></li>
            <?php endforeach; ?>
        </ul>
    <?php elseif (isset($error)): ?>
        <p><?= htmlspecialchars($error) ?></p>
    <?php endif; ?>
</body>
</html>
